## 1.BM1反转链表

```c++
/*
struct ListNode {
	int val;
	struct ListNode *next;
	ListNode(int x) :
			val(x), next(NULL) {
	}
};*/
class Solution {
public:
    ListNode* ReverseList(ListNode* pHead) {
        if(pHead==NULL||pHead->next==NULL){
            return pHead;
        }
        stack<ListNode*>stk;
        ListNode*temp=pHead;
        while(temp!=NULL){
            stk.push(temp);
            temp=temp->next;
        }
        ListNode*temp2=stk.top();
        //ListNode*temp2=NULL;
        ListNode*reverseHead=temp2;//由于reverseHead和temp2指针指向同一块地址，而while循环里，                                        若改变temp2，改变的只是它的指向，而不是原先指向地址的内容，所以                                      reverseHead仍然指向原先的地址，并不会做任何调整。所以为了让                                        reverseHead不丢失temp2，我们最好只改变temp->next指针的指                                        向，这样reverseHead就不会丢失temp2
        stk.pop();
        while(stk.size()>0){
            ListNode* tempNode=stk.top();
            temp2->next=tempNode;
            //temp2=stk.top();
            stk.pop();
            temp2=temp2->next;
        }
        temp2->next=NULL;//防止构成一个环形单链表
        return reverseHead;
    }
};
```

```c++
/**
 * struct ListNode {
 *	int val;
 *	struct ListNode *next;
 * };
 */

class Solution {
public:
    /**
     * 
     * @param head ListNode类 
     * @param m int整型 
     * @param n int整型 
     * @return ListNode类
     */
    ListNode* reverseBetween(ListNode* head, int m, int n) {
        ListNode* dummyNode=new ListNode(-1);//虚拟头节点十分重要，因为在众多变化的情况下，要想找到                                                原来的完整链表，就需要有一个永远不变的“寻回”
        dummyNode->next=head;
        ListNode*pre=dummyNode;
        for(int i=1;i<m;i++){
            pre=pre->next;
        }
        ListNode*left=pre->next;
        pre->next=NULL;
        ListNode*right=left;
        for(int j=m;j<n;j++){
            right=right->next;
        }
        ListNode*cur=right->next;
        right->next=NULL;
        reverseLinkedList(left);
        pre->next=right;
        left->next=cur;
        return dummyNode->next;//注意这里不要把dummy->next当作head，因为如果指定区间包含head指向的                                节点，那么经过反转后，head将会到原先right所在位，而不是原来的头节点了
    }
    
    private:
    void reverseLinkedList(ListNode*head){
        ListNode*reverseHead=NULL;
        ListNode*cur=head;
        while(cur!=NULL){
            ListNode* next=cur->next;
            cur->next=reverseHead;
            reverseHead=cur;
            cur=next;
        }
    }
};
```

## 3.BM3链表中的节点每k个一组反转

```c++
方法一：链表分块反转
/**
 * struct ListNode {
 *	int val;
 *	struct ListNode *next;
 * };
 */

class Solution {
public:
    /**
     * 
     * @param head ListNode类 
     * @param k int整型 
     * @return ListNode类
     */
    ListNode* reverseKGroup(ListNode* head, int k) {
        // write code here
        if(k<=1||head==NULL||head->next==NULL)return head;
        int len=getLength(head);
        int sx=len/k;
        ListNode*result=new ListNode(-1);
        ListNode*cur=result;
        int cnt=0;
        for(int i=0;i<sx;i++){
            ListNode*temp=NULL;
            for(int j=0;j<k;j++){
                ListNode*next=head->next;
                head->next=temp;
                temp=head;
                head=next;
            }
            cur->next=temp;
            while(cur->next!=NULL){
                cur=cur->next;
            }
        }
        cur->next=head;
        return result->next;
    }
    
private:
    int getLength(ListNode*head){
        int res=0;
        while(head!=NULL){
            res++;
            head=head->next;
        }
        return res;
    }
};

方法二：链表分块放入栈

class Solution {
public:
    /**
     * 
     * @param head ListNode类 
     * @param k int整型 
     * @return ListNode类
     */
    ListNode* reverseKGroup(ListNode* head, int k) {
        // write code here
        if(k==1||head==NULL||head->next==NULL){
            return head;
        }
        stack<ListNode*>stk;
        ListNode*result=new ListNode(-1);
        ListNode*cur=result;
        while(true){
            int cnt=0;//用于判断栈中是否有k个元素
            for(int i=0;i<k;i++){
                stk.push(head);
                head=head->next;
                cnt++;
                if(head==NULL)break;
            }
            if(cnt==k){
                while(!stk.empty()){
                    cur->next=stk.top();
                    stk.pop();
                    cur=cur->next;
                    cur->next=NULL;
                }
            }
            if(head==NULL)break;
        }
        ListNode*end=NULL;
        while(!stk.empty()){
            end=stk.top();
            stk.pop();
        }
        cur->next=end;
        return result->next;
    }
};
```

## BM4合并两个排序链表

```c++
/*
struct ListNode {
	int val;
	struct ListNode *next;
	ListNode(int x) :
			val(x), next(NULL) {
	}
};*/
class Solution {
public:
    ListNode* Merge(ListNode* pHead1, ListNode* pHead2) {
        ListNode*dummyHead=new ListNode(-1);
        ListNode*resHead=dummyHead;
        while(pHead1&&pHead2){
            if(pHead1->val<=pHead2->val){
                dummyHead->next=pHead1;
                pHead1=pHead1->next;
            }
            else{
                dummyHead->next=pHead2;
                pHead2=pHead2->next;
            }
            dummyHead=dummyHead->next;
        }
        if(!pHead1){
            dummyHead->next=pHead2;
        }
        if(!pHead2){
            dummyHead->next=pHead1;
        }
        return resHead->next;
    }
};
```

## BM5合并k个已排序的链表

```c++
正确代码：
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode*mergeTwoLists(ListNode*l1,ListNode*l2){
        if(l1==NULL||l2==NULL){
            return l1?l1:l2;
        }
        ListNode*resNode=new ListNode(-1);
        ListNode*cur=resNode;
        ListNode*temp1=l1;
        ListNode*temp2=l2;
        /*ListNode*resNode=new ListNode(-1);
        ListNode*cur=resNode;*/
        //由于对链表容器是是引用，最好使用临时指针代替头指针，因为这样做不会改变原容器的内容，避免造成一些           无法预料的错误
        while(temp1&&temp2){
            if(temp1->val<=temp2->val){
                cur->next=temp1;
                temp1=temp1->next;
            }
            else{//if(temp2->val<=temp1->val)
                 //按理来说，每一个while循环理应只让resNode链表添加一个节点，但如果用两个if的话，就会                    判断两次，会导致cur还未后移时，又让cur->next指向另一个节点，导致紊乱
                cur->next=temp2;
                temp2=temp2->next;
            }
            cur=cur->next;
        }
        cur->next=(temp1?temp1:temp2);
        return resNode->next;
    }
    
    ListNode*merge(vector<ListNode*>&lists,int l,int r){
        if(l==r)return lists[l];
        if(l>r)return NULL;
        int mid=l+((r-l)>>1);
        return mergeTwoLists(merge(lists,l,mid), merge(lists,mid+1,r));
    }
    
    ListNode *mergeKLists(vector<ListNode *> &lists) {
        return merge(lists,0,lists.size()-1);
    }
};
```

## BM6判断链表中是否有环

```c++
方法一：快慢指针
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    bool hasCycle(ListNode *head) {
        if(head->next==NULL||head==NULL){//注意只有两个节点及以上才能构成环
            return false;
        }
        ListNode*fast=head;
        ListNode*slow=head;
        while(fast){
            slow=slow->next;
            if(fast->next){//由于fast行动过快，可能会导致它指向空节点的下一个节点，所以我们要判断一下                              fast的下一个节点是否为空，如果不为空,那么就可以跳两个；如果为空，说明不                              成环，直接return false
            fast=fast->next->next;
            }else{
                return false;
            }
            if(fast==slow){
                return true;
            }
        }
        return false;
    }
};

方法二：哈希表
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    bool hasCycle(ListNode *head) {
        set<ListNode*>st;
        while(head!=NULL){
            if(st.find(head)!=st.end()){
                return true;
            }
            st.insert(head);
            head=head->next;
        }
        return false;
    }
};
```

```c++
/*
struct ListNode {
    int val;
    struct ListNode *next;
    ListNode(int x) :
        val(x), next(NULL) {
    }
};
*/
class Solution {
public:
    ListNode* EntryNodeOfLoop(ListNode* pHead) {
        if(!Circle(pHead)){
            return NULL;
        }
        ListNode*fast=pHead;
        ListNode*slow=pHead;
        while(fast){
            fast=fast->next->next;
            slow=slow->next;
            if(fast==slow){
                break;
            }
        }
        fast=pHead;
        while(fast!=slow){
            slow=slow->next;
            fast=fast->next;
        }
        return fast;
    }
    
    bool Circle(ListNode*head){
        if(head==NULL||head->next==NULL){
            return false;
        }
        ListNode*fast=head;
        ListNode*slow=head;
        while(fast){
            if(fast->next){
                fast=fast->next->next;
            }else{
                return false;
            }
            slow=slow->next;
            if(fast==slow){
                return true;
            }
        }
        return false;
    }
};
```

## BM8链表中倒数最后k个节点

```c++
/**
 * struct ListNode {
 *	int val;
 *	struct ListNode *next;
 *	ListNode(int x) : val(x), next(nullptr) {}
 * };
 */
class Solution {
public:
    /**
     * 代码中的类名、方法名、参数名已经指定，请勿修改，直接返回方法规定的值即可
     *
     * 
     * @param pHead ListNode类 
     * @param k int整型 
     * @return ListNode类
     */
    ListNode* FindKthToTail(ListNode* pHead, int k) {
        // write code here
        ListNode*temp=pHead;
        int length=0;
        while(temp!=NULL){
            length++;
            temp=temp->next;
        }
        if(k>length){
            return NULL;
        }
        int dis=length-k;
        ListNode*res=pHead;
        while(dis>0){//注意不要多移一步
            res=res->next;
            dis--;
        }
        return res;
    }
};
```

## BM9删除链表的倒数第n个节点

```c++
/**
 * struct ListNode {
 *	int val;
 *	struct ListNode *next;
 * };
 */

class Solution {
public:
    /**
     * 
     * @param head ListNode类 
     * @param n int整型 
     * @return ListNode类
     */
    ListNode* removeNthFromEnd(ListNode* head, int n) {
        // write code here
        ListNode*temp=head;
        int length=0;
        while(temp){
            length++;
            temp=temp->next;
        }
        if(length==n){
            return head->next;
        }
        int pre=length-n-1;
        ListNode*resHead=head;
        ListNode*cur=resHead;
        while(pre>0){
            cur=cur->next;
            pre--;
        }
        cur->next=cur->next->next;
        return resHead;
    }
};
```

## BM10两个链表的第一个公共节点

```c++
/*
struct ListNode {
	int val;
	struct ListNode *next;
	ListNode(int x) :
			val(x), next(NULL) {
	}
};*/
class Solution {
public:
    ListNode* FindFirstCommonNode( ListNode* pHead1, ListNode* pHead2) {
        if(!pHead1||!pHead2){
            return NULL;
        }
        
        //计算两链表的长度
        int length1=0;
        int length2=0;
        ListNode*temp1=pHead1;
        ListNode*temp2=pHead2;
        while(temp1->next!=NULL){
            temp1=temp1->next;
            length1++;
        }
        while(temp2->next!=NULL){
            temp2=temp2->next;
            length2++;
        }
        //如果temp1!=temp2，说明两链表的尾节点不一样，所以就没有公共节点了
        if(temp1!=temp2){
            return NULL;
        }
        
        //让长的链表先移动
        int length=0;
        if(length1>length2){
            length=length1-length2;
            while(length>0){
                pHead1=pHead1->next;
                length--;
            }
        }else{
            length=length2-length1;
            while(length>0){
                pHead2=pHead2->next;
                length--;
            }
        }
        
        //两指针到公共节点的距离相同，故只要一直后移，直到相等即可
        while(pHead1!=pHead2){
            pHead1=pHead1->next;
            pHead2=pHead2->next;
        }
        return pHead1;
    }
};
```
