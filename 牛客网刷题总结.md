## 1.BM1反转链表

```c++
/*
struct ListNode {
	int val;
	struct ListNode *next;
	ListNode(int x) :
			val(x), next(NULL) {
	}
};*/
class Solution {
public:
    ListNode* ReverseList(ListNode* pHead) {
        if(pHead==NULL||pHead->next==NULL){
            return pHead;
        }
        stack<ListNode*>stk;
        ListNode*temp=pHead;
        while(temp!=NULL){
            stk.push(temp);
            temp=temp->next;
        }
        ListNode*temp2=stk.top();
        //ListNode*temp2=NULL;
        ListNode*reverseHead=temp2;//由于reverseHead和temp2指针指向同一块地址，而while循环里，                                        若改变temp2，改变的只是它的指向，而不是原先指向地址的内容，所以                                      reverseHead仍然指向原先的地址，并不会做任何调整。所以为了让                                        reverseHead不丢失temp2，我们最好只改变temp->next指针的指                                        向，这样reverseHead就不会丢失temp2
        stk.pop();
        while(stk.size()>0){
            ListNode* tempNode=stk.top();
            temp2->next=tempNode;
            //temp2=stk.top();
            stk.pop();
            temp2=temp2->next;
        }
        temp2->next=NULL;//防止构成一个环形单链表
        return reverseHead;
    }
};
```

```c++
/**
 * struct ListNode {
 *	int val;
 *	struct ListNode *next;
 * };
 */

class Solution {
public:
    /**
     * 
     * @param head ListNode类 
     * @param m int整型 
     * @param n int整型 
     * @return ListNode类
     */
    ListNode* reverseBetween(ListNode* head, int m, int n) {
        ListNode* dummyNode=new ListNode(-1);//虚拟头节点十分重要，因为在众多变化的情况下，要想找到                                                原来的完整链表，就需要有一个永远不变的“寻回”
        dummyNode->next=head;
        ListNode*pre=dummyNode;
        for(int i=1;i<m;i++){
            pre=pre->next;
        }
        ListNode*left=pre->next;
        pre->next=NULL;
        ListNode*right=left;
        for(int j=m;j<n;j++){
            right=right->next;
        }
        ListNode*cur=right->next;
        right->next=NULL;
        reverseLinkedList(left);
        pre->next=right;
        left->next=cur;
        return dummyNode->next;//注意这里不要把dummy->next当作head，因为如果指定区间包含head指向的                                节点，那么经过反转后，head将会到原先right所在位，而不是原来的头节点了
    }
    
    private:
    void reverseLinkedList(ListNode*head){
        ListNode*reverseHead=NULL;
        ListNode*cur=head;
        while(cur!=NULL){
            ListNode* next=cur->next;
            cur->next=reverseHead;
            reverseHead=cur;
            cur=next;
        }
    }
};
```
