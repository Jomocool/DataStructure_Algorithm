## 1.BM1反转链表

```c++
/*
struct ListNode {
	int val;
	struct ListNode *next;
	ListNode(int x) :
			val(x), next(NULL) {
	}
};*/
class Solution {
public:
    ListNode* ReverseList(ListNode* pHead) {
        if(pHead==NULL||pHead->next==NULL){
            return pHead;
        }
        stack<ListNode*>stk;
        ListNode*temp=pHead;
        while(temp!=NULL){
            stk.push(temp);
            temp=temp->next;
        }
        ListNode*temp2=stk.top();
        //ListNode*temp2=NULL;
        ListNode*reverseHead=temp2;//由于reverseHead和temp2指针指向同一块地址，而while循环里，                                        若改变temp2，改变的只是它的指向，而不是原先指向地址的内容，所以                                      reverseHead仍然指向原先的地址，并不会做任何调整。所以为了让                                        reverseHead不丢失temp2，我们最好只改变temp->next指针的指                                        向，这样reverseHead就不会丢失temp2
        stk.pop();
        while(stk.size()>0){
            ListNode* tempNode=stk.top();
            temp2->next=tempNode;
            //temp2=stk.top();
            stk.pop();
            temp2=temp2->next;
        }
        temp2->next=NULL;//防止构成一个环形单链表
        return reverseHead;
    }
};
```

```c++
        right->next=NULL;

        pre->next=right;
        left->next=cur;
        return dummyNode->next;//注意这里不要把dummy->next当作head，因为如果指定区间包含head指向的                                节点，那么经过反转后，head将会到原先right所在位，而不是原来的头节点了
    }
    
    private:
    void reverseLinkedList(ListNode*head){
        ListNode*reverseHead=NULL;
        ListNode*cur=head;
        while(cur!=NULL){
            ListNode* next=cur->next;
            cur->next=reverseHead;
            reverseHead=cur;
            cur=next;
        }
    }
};
```

## 3.BM3链表中的节点每k个一组反转

```c++
方法一：链表分块反转
/**

 *	int val;

class Solution {
public:
    /**
     * 
     * @param head ListNode类 
     * @param k int整型 
     * @return ListNode类
     */
    ListNode* reverseKGroup(ListNode* head, int k) {
        // write code here
        if(k<=1||head==NULL||head->next==NULL)return head;
        int len=getLength(head);
        int sx=len/k;
        ListNode*result=new ListNode(-1);
        ListNode*cur=result;
        int cnt=0;
        for(int i=0;i<sx;i++){
            ListNode*temp=NULL;
            for(int j=0;j<k;j++){
                ListNode*next=head->next;
                head->next=temp;
                temp=head;
                head=next;
            }
            cur->next=temp;
            while(cur->next!=NULL){
                cur=cur->next;
            }
        }
        cur->next=head;
        return result->next;
    }
    
private:
    int getLength(ListNode*head){
        int res=0;
        while(head!=NULL){
            res++;
            head=head->next;
        }
        return res;
    }
};

        while(!stk.empty()){
            end=stk.top();
            stk.pop();
        }
        cur->next=end;
        return result->next;
    }
};
```

```c++
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    ListNode*mergeTwoLists(ListNode*l1,ListNode*l2){
        if(l1==NULL||l2==NULL){
            return l1?l1:l2;
        }
        ListNode*resNode=new ListNode(-1);
        ListNode*cur=resNode;
        ListNode*temp1=l1;
        ListNode*temp2=l2;
        /*ListNode*resNode=new ListNode(-1);
        ListNode*cur=resNode;*/
        //由于对链表容器是是引用，最好使用临时指针代替头指针，因为这样做不会改变原容器的内容，避免造成一些           无法预料的错误
        while(temp1&&temp2){
            if(temp1->val<=temp2->val){
                cur->next=temp1;
                temp1=temp1->next;
            }
            else{//if(temp2->val<=temp1->val)
                 //按理来说，每一个while循环理应只让resNode链表添加一个节点，但如果用两个if的话，就会                    判断两次，会导致cur还未后移时，又让cur->next指向另一个节点，导致紊乱
                cur->next=temp2;
                temp2=temp2->next;
            }
            cur=cur->next;
        }
        cur->next=(temp1?temp1:temp2);
        return resNode->next;
    }
    
    ListNode*merge(vector<ListNode*>&lists,int l,int r){
        if(l==r)return lists[l];
        if(l>r)return NULL;
        int mid=l+((r-l)>>1);
        return mergeTwoLists(merge(lists,l,mid), merge(lists,mid+1,r));
    }
    
    ListNode *mergeKLists(vector<ListNode *> &lists) {
        return merge(lists,0,lists.size()-1);
    }
};
```

## BM6判断链表中是否有环

```c++
方法一：快慢指针
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    bool hasCycle(ListNode *head) {
        if(head->next==NULL||head==NULL){//注意只有两个节点及以上才能构成环
            return false;
        }
        ListNode*fast=head;
        ListNode*slow=head;
        while(fast){
            slow=slow->next;
            if(fast->next){//由于fast行动过快，可能会导致它指向空节点的下一个节点，所以我们要判断一下                              fast的下一个节点是否为空，如果不为空,那么就可以跳两个；如果为空，说明不                              成环，直接return false
            fast=fast->next->next;
            }else{
                return false;
            }
            if(fast==slow){
                return true;
            }
        }
        return false;
    }
};

方法二：哈希表
/**
 * Definition for singly-linked list.
 * struct ListNode {
 *     int val;
 *     ListNode *next;
 *     ListNode(int x) : val(x), next(NULL) {}
 * };
 */
class Solution {
public:
    bool hasCycle(ListNode *head) {
        set<ListNode*>st;
        while(head!=NULL){
            if(st.find(head)!=st.end()){
                return true;
            }
            st.insert(head);
            head=head->next;
        }
        return false;
    }
};
```

```c++
        if(!Circle(pHead)){
            return NULL;
        }

            fast=fast->next->next;
            slow=slow->next;

            return false;
        }

        ListNode*slow=head;
        while(fast){
            if(fast->next){
                fast=fast->next->next;
            }else{
                return false;
            }
            slow=slow->next;
            if(fast==slow){
                return true;
            }
        }
        return false;
    }
};
```

## BM10两个链表的第一个公共节点

```c++
/*
struct ListNode {
	int val;
	struct ListNode *next;
	ListNode(int x) :
			val(x), next(NULL) {
	}
};*/
class Solution {
public:
    ListNode* FindFirstCommonNode( ListNode* pHead1, ListNode* pHead2) {
        if(!pHead1||!pHead2){
            return NULL;
        }
        
        //计算两链表的长度
        int length1=0;
        int length2=0;
        ListNode*temp1=pHead1;
        ListNode*temp2=pHead2;
        while(temp1->next!=NULL){
            temp1=temp1->next;
            length1++;
        }
        while(temp2->next!=NULL){
            temp2=temp2->next;
            length2++;
        }
        //如果temp1!=temp2，说明两链表的尾节点不一样，所以就没有公共节点了
        if(temp1!=temp2){
            return NULL;
        }
        
        //让长的链表先移动
        int length=0;
        if(length1>length2){
            length=length1-length2;
            while(length>0){
                pHead1=pHead1->next;
                length--;
            }
        }else{
            length=length2-length1;
            while(length>0){
                pHead2=pHead2->next;
                length--;
            }
        }
        
        //两指针到公共节点的距离相同，故只要一直后移，直到相等即可
        while(pHead1!=pHead2){
            pHead1=pHead1->next;
            pHead2=pHead2->next;
        }
        return pHead1;
    }
};
```

## BM12单链表的排序

```c++
方法：归并
关键：不断二分链表，再不断合并
/**
 * struct ListNode {
 *	int val;
 *	struct ListNode *next;
 * };
 */

class Solution {
public:
    /**
     * 
     * @param head ListNode类 the head node
     * @return ListNode类
     */
    ListNode* sortInList(ListNode* head) {
        // write code here
        if(head==NULL||head->next==NULL){
            return head;
        }
        ListNode*left=head;
        ListNode*mid=head->next;
        ListNode*right=head->next->next;
        while(right!=NULL&&right->next!=NULL){
            left=left->next;
            mid=mid->next;
            right=right->next->next;
        }
        left->next=NULL;
        return merge(sortInList(head),sortInList(mid));
    }
    
    ListNode* merge(ListNode* pHead1, ListNode* pHead2) {
        ListNode*dummyHead=new ListNode(-1);
        ListNode*resHead=dummyHead;
        while(pHead1&&pHead2){
            if(pHead1->val<=pHead2->val){
                dummyHead->next=pHead1;
                pHead1=pHead1->next;
            }
            else{
                dummyHead->next=pHead2;
                pHead2=pHead2->next;
            }
            dummyHead=dummyHead->next;
        }
        if(!pHead1){
            dummyHead->next=pHead2;
        }
        if(!pHead2){
            dummyHead->next=pHead1;
        }
        return resHead->next;
    }
};
```

## BM13判断一个链表是否为回文结构

```c++
方法：快慢指针
关键在于理清链表长度分别为奇偶的操作：
    1.如果是奇数，fast刚好到最后一个节点，而slow在正中间的节点，因为回文结构共享正中间的数，所以正中间的       节点不用管，slow后移一个节点
    2.如果是偶数，fast到最后一个节点的next，slow刚好到了反转位置节点，无需更多操作
/**
 * struct ListNode {
 *	int val;
 *	struct ListNode *next;
 * };
 */

class Solution {
public:
    /**
     * 
     * @param head ListNode类 the head
     * @return bool布尔型
     */
    bool isPail(ListNode* head) {
        // write code here
        ListNode*fast=head;
        ListNode*slow=head;
        while(fast!=NULL&&fast->next!=NULL){
            fast=fast->next->next;
            slow=slow->next;
        }
        //如果fast不为空，说明链表的长度是奇数
        if(fast!=NULL){
            slow=slow->next;
        }
        //反转后半部分链表
        slow=reverse(slow);
        while(slow){
            if(slow->val!=head->val)return false;
            slow=slow->next;
            head=head->next;
        }
        return true;
    }
    
    ListNode* reverse(ListNode*head){
        if(head==NULL||head->next==NULL){
            return head;
        }
        ListNode*reverseHead=new ListNode(-1);
        ListNode*cur=head;
        ListNode*next=NULL;

            next=cur->next;
            cur->next=reverseHead->next;
            reverseHead->next=cur;
            cur=next;
        }
        return reverseHead->next;
    }
};
```

## BM20数组中的逆序对

```c++
class Solution {
  public:
    int count = 0;

    int InversePairs(vector<int> data) {
        if(data.size()<2){
            return 0;
        }
        mergeSort(data,0,data.size()-1);
        return count;
    }

    void merge(vector<int>& vec, int left, int mid, int right) {
        vector<int> temp(right - left + 1);
        int tempIndex = 0;
        int l = left;   //左侧指针
        int r = mid + 1;    //右侧指针
        //越界时跳出
        while (l <= mid && r <= right) {
            if(vec[l]<=vec[r]){
                temp[tempIndex]=vec[l];
                tempIndex++;
                l++;
            }else{
                temp[tempIndex]=vec[r];
                tempIndex++;
                r++;
                count+=mid+1-l;//两边都必然有序，右边的值小于左边的值，那么也必然小于左边后面的值
                count%=1000000007;
            }
        }
        //拷贝剩余的数据到辅助数组
        while (l <= mid) {
            temp[tempIndex++] = vec[l++];
        }
        while (r <= right) {
            temp[tempIndex++] = vec[r++];
        }
        //数据拷贝
        for (int i = 0; i < temp.size(); i++) {
            vec[left + i] = temp[i];
        }
    }

    void mergeSort(vector<int>& vec, int left, int right) {
        if (vec.size() < 2) {
            return;
        }

        if (left < right) {
            int mid = left + ((right - left) >> 1);
            mergeSort(vec, left, mid);
            mergeSort(vec, mid + 1, right);
            merge(vec, left, mid, right);
        }
    }
};
```
